150  std::deque<Tile*> openSet;

153  //push back the starting tile
154  openSet.push_back(getTile(start.x, start.y));
155  //the start tile has no parent
156  parent[getTile(start.x, start.y)] = NULL;
157  //the end tile

159  //as long as the end tile has no parent
160  while(parent.count(endTile) == 0)
161  {
162  //if there are no tiles in the openSet then there is no path
163  if(openSet.empty())

172  const int yChange[] = {-1, 1, 0, 0};
173  //look in all directions
174  for(unsigned i = 0; i < 4; ++i)
175  {
176  Point loc(curTile->x() + xChange[i], curTile->y() + yChange[i]);
177  Tile* toAdd = getTile(loc.x, loc.y);
178  //if a tile exists there
179  if(toAdd != NULL)
180  {
181  //if it's an open tile and it doesn't have a parent
182  if(toAdd->type() == Tile::EMPTY && parent.count(toAdd) == 0)
183  {

185  openSet.push_back(toAdd);
